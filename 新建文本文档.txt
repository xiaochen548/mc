<script>
    const canvas = document.getElementById('bgCanvas');
    const ctx = canvas.getContext('2d');
    const themeBtn = document.getElementById('themeBtn');
    const body = document.getElementById('body');
    const footer = document.getElementById('footer');
    const cardBg = document.querySelectorAll('.card-bg');
    const sectionTitles = document.querySelectorAll('.section-title');
    const searchModal = document.getElementById('searchModal');
    const modalContent = document.getElementById('modalContent');
    const timelineTitles = document.querySelectorAll('.timeline-title');
    const timelineTexts = document.querySelectorAll('.timeline-text');

    let isDark = true;
    let animationId = null;
    let activeTooltip = null;
    let isHovering = false;
    let hideTimeout = null;
    let currentCopyText = '';

    // 关闭搜索弹窗
    function openModal() {
        searchModal.style.display = "flex";
        document.body.style.overflow = "hidden";
    }
    function closeModal() {
        searchModal.style.display = "none";
        document.body.style.overflow = "auto";
    }
    searchModal.addEventListener('click', (e) => {
        if(e.target === searchModal) closeModal();
    });

    // 主题切换
    themeBtn.addEventListener('click', () => {
        isDark = !isDark;
        if (isDark) {
            body.className = 'bg-[#0a101f] text-gray-200';
            themeBtn.className = 'theme-toggle bg-gray-800 text-white shadow-md';
            themeBtn.innerHTML = '<i class="fas fa-moon"></i> 切换亮色';
            sectionTitles.forEach(el => el.classList.remove('text-black'));
            sectionTitles.forEach(el => el.classList.add('text-white'));
            footer.className = 'text-center text-gray-400 text-sm pt-6 border-t border-gray-700';
            cardBg.forEach(el => el.classList.remove('bg-white'));
            cardBg.forEach(el => el.classList.add('bg-gray-800'));
            timelineTitles.forEach(el => el.classList.remove('text-black'));
            timelineTitles.forEach(el => el.classList.add('text-white'));
            timelineTexts.forEach(el => el.classList.remove('text-black'));
            timelineTexts.forEach(el => el.classList.add('text-gray-300'));
            modalContent.className = 'search-modal bg-gray-800 text-white';
        } else {
            body.className = 'bg-gray-100 text-black';
            themeBtn.className = 'theme-toggle bg-white text-black shadow-md';
            themeBtn.innerHTML = '<i class="fas fa-sun"></i> 切换暗色';
            sectionTitles.forEach(el => el.classList.remove('text-white'));
            sectionTitles.forEach(el => el.classList.add('text-black'));
            footer.className = 'text-center text-black text-sm pt-6 border-t border-gray-200';
            cardBg.forEach(el => el.classList.remove('bg-gray-800'));
            cardBg.forEach(el => el.classList.add('bg-white'));
            timelineTitles.forEach(el => el.classList.remove('text-white'));
            timelineTitles.forEach(el => el.classList.add('text-black'));
            timelineTexts.forEach(el => el.classList.remove('text-gray-300'));
            timelineTexts.forEach(el => el.classList.add('text-black'));
            modalContent.className = 'search-modal bg-white text-black';
        }
    });

    // 画布适配
    function resizeCanvas() {
        canvas.width = window.innerWidth;
        canvas.height = window.innerHeight;
    }
    resizeCanvas();
    window.addEventListener('resize', resizeCanvas);

    // MC指令动画
    const mcCommands = [
        "/give @s apple 1","/tp @s ~ ~10 ~","/summon zombie ~ ~ ~",
        "/kill @e[type=item]","/clear @s","/setblock ~ ~1 ~ stone",
        "/gamemode creative","/gamerule keepInventory true","/effect give @s speed 10 1",
        "/teleport @p village","/give @s diamond_sword 1","/time set day","/weather clear"
    ];
    const commands = [];
    const COMMAND_SPEED = 2.2;
    const MIN_VERTICAL_GAP = 130;

    function spawnCommand() {
        let y;
        let attempts = 0;
        do {
            y = Math.random() * (canvas.height - 100) + 50;
            attempts++;
        } while (commands.some(cmd => Math.abs(cmd.y - y) < MIN_VERTICAL_GAP) && attempts < 25);
        commands.push({ x: -180, y: y, text: mcCommands[Math.floor(Math.random() * mcCommands.length)] });
    }

    function drawCommands() {
        ctx.clearRect(0, 0, canvas.width, canvas.height);
        ctx.font = '22px KaiTi, serif';
        ctx.textBaseline = "middle";
        ctx.shadowBlur = 8;

        for (let i = commands.length - 1; i >= 0; i--) {
            const cmd = commands[i];
            const textWidth = ctx.measureText(cmd.text).width;
            const padding = 12; const bgHeight = 28;

            ctx.fillStyle = "rgba(0, 0, 0, 0.25)";
            ctx.fillRect(cmd.x - padding/2, cmd.y - bgHeight/2, textWidth + padding, bgHeight);
            ctx.fillStyle = isDark ? "#ffffff" : "#000000";
            ctx.shadowColor = isDark ? "rgba(255,255,255,0.3)" : "rgba(0,0,0,0.2)";
            
            ctx.fillText(cmd.text, cmd.x, cmd.y);
            cmd.x += COMMAND_SPEED;
            if (cmd.x > canvas.width + 200) commands.splice(i, 1);
        }
        if (Math.random() < 0.013) spawnCommand();
        animationId = requestAnimationFrame(drawCommands);
    }
    drawCommands();

    // ====================== 新版气泡功能（鼠标悬浮+点击常驻+复制按钮+6秒消失）======================
    function removeTooltip() {
        if (activeTooltip) {
            activeTooltip.remove();
            activeTooltip = null;
        }
        if (hideTimeout) {
            clearTimeout(hideTimeout);
            hideTimeout = null;
        }
    }

    function copyToClipboard() {
        navigator.clipboard.writeText(currentCopyText).then(() => {
            alert("复制成功！");
        });
    }

    function showTooltip(element, text, copyText) {
        currentCopyText = copyText;
        removeTooltip();
        isHovering = true;

        const bubble = document.createElement('div');
        bubble.className = 'tooltip-bubble';
        
        // 主题适配
        if (isDark) {
            bubble.style.background = '#1f2937';
            bubble.style.color = '#fff';
        } else {
            bubble.style.background = '#ffffff';
            bubble.style.color = '#000';
        }

        // 气泡内容 + 复制按钮
        bubble.innerHTML = `
            <span>${text}</span>
            <div class="copy-btn bg-blue-500 text-white">复制</div>
        `;

    // 鼠标离开 → 延迟6秒消失
    function hideTooltipDelay() {
        isHovering = false;
        if (hideTimeout) clearTimeout(hideTimeout);
        hideTimeout = setTimeout(() => {
            if (!isHovering && activeTooltip) {
                activeTooltip.classList.add('tooltip-hide');
                setTimeout(removeTooltip, 1000);
            }
        }, 6000);
    }

    // 点击 → 强制显示6秒
    function showTooltipFixed() {
        if (hideTimeout) clearTimeout(hideTimeout);
        hideTimeout = setTimeout(() => {
            if (activeTooltip) {
                activeTooltip.classList.add('tooltip-hide');
                setTimeout(removeTooltip, 1000);
            }
        }, 6000);
    }
