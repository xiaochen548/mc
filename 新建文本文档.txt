    <script>
        const canvas = document.getElementById('bgCanvas');
        const ctx = canvas.getContext('2d');
        let isDark = true;

        // 主题切换
        const themeBtn = document.getElementById('themeBtn');
        const body = document.getElementById('body');
        const navTitle = document.getElementById('navTitle');
        const careerTitle = document.getElementById('careerTitle');
        const galleryTitle = document.getElementById('galleryTitle');
        const footer = document.getElementById('footer');
        const cards = document.querySelectorAll('.bg-white');

        themeBtn.addEventListener('click', () => {
            isDark = !isDark;
            if (isDark) {
                body.className = 'bg-[#0a101f] text-gray-700';
                themeBtn.className = 'theme-toggle bg-white text-gray-800 shadow-md';
                themeBtn.innerHTML = '<i class="fas fa-moon"></i> 切换亮色';
                navTitle.className = 'text-xl font-bold mb-6 flex items-center text-white';
                careerTitle.className = 'text-xl font-bold mb-6 flex items-center text-white';
                galleryTitle.className = 'text-xl font-bold mb-6 flex items-center text-white';
                footer.className = 'text-center text-gray-300 text-sm pt-6 border-t border-gray-700';
                cards.forEach(c => c.className = 'bg-white rounded-2xl p-8 shadow-sm mb-12 card-hover');
            } else {
                body.className = 'bg-white text-gray-800';
                themeBtn.className = 'theme-toggle bg-gray-800 text-white shadow-md';
                themeBtn.innerHTML = '<i class="fas fa-sun"></i> 切换暗色';
                navTitle.className = 'text-xl font-bold mb-6 flex items-center text-gray-800';
                careerTitle.className = 'text-xl font-bold mb-6 flex items-center text-gray-800';
                galleryTitle.className = 'text-xl font-bold mb-6 flex items-center text-gray-800';
                footer.className = 'text-center text-gray-500 text-sm pt-6 border-t border-gray-200';
                cards.forEach(c => c.className = 'bg-[#f7f7fb] rounded-2xl p-8 shadow-sm mb-12 card-hover');
            }
        });

        function resizeCanvas() {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
        }
        resizeCanvas();
        window.addEventListener('resize', resizeCanvas);

        const mcCommands = [
            "/give @s apple 1","/tp @s ~ ~10 ~","/summon zombie ~ ~ ~",
            "/kill @e[type=item]","/clear @s","/setblock ~ ~1 ~ stone",
            "/gamemode creative","/gamerule keepInventory true","/effect give @s speed 10 1",
            "/teleport @p village","/give @s diamond_sword 1","/time set day","/weather clear"
        ];

        const commands = [];
        const speed = 1.3;
        const minDistance = 120;

        function isOverlap(y) {
            for (let c of commands) {
                if (Math.abs(c.y - y) < minDistance) return true;
            }
            return false;
        }

        function spawnCommand() {
            let y = Math.random() * canvas.height;
            let tries = 0;
            while (isOverlap(y) && tries < 20) {
                y = Math.random() * canvas.height;
                tries++;
            }
            if (tries >= 20) return;
            commands.push({
                x: -200, y: y,
                text: mcCommands[Math.floor(Math.random() * mcCommands.length)],
                speed: speed + Math.random() * 0.4
            });
        }

        function draw() {
            if (isDark) {
                ctx.fillStyle = 'rgba(10,16,31,0.08)';
            } else {
                ctx.fillStyle = 'rgba(255,255,255,0.08)';
            }
            ctx.fillRect(0,0,canvas.width,canvas.height);

            ctx.font = '22px "KaiTi", "STKaiti", serif';
            ctx.textBaseline = "middle";

            for (let i=commands.length-1; i>=0; i--) {
                const cmd = commands[i];
                const w = ctx.measureText(cmd.text).width;
                const h = 28;

                if (isDark) {
                    ctx.fillStyle = 'rgba(0,0,0,0.5)';
                    ctx.fillRect(cmd.x-6, cmd.y-h/2, w+12, h);
                    ctx.shadowColor = "#fff";
                    ctx.shadowBlur = 6;
                    ctx.fillStyle = "#fff";
                } else {
                    ctx.fillStyle = 'rgba(255,255,255,0.5)';
                    ctx.fillRect(cmd.x-6, cmd.y-h/2, w+12, h);
                    ctx.shadowColor = "#000";
                    ctx.shadowBlur = 6;
                    ctx.fillStyle = "#000";
                }

                ctx.fillText(cmd.text, cmd.x, cmd.y);
                ctx.shadowBlur = 0;
                cmd.x += cmd.speed;
                if (cmd.x > canvas.width + 200) commands.splice(i,1);
            }

            if (Math.random() < 0.014) spawnCommand();
        }

        setInterval(draw, 22);
    </script>
